name: AI Detection

on:
  pull_request:
    types: [opened, edited, synchronize]

jobs:
  detect-ai:
    runs-on: ubuntu-latest
    steps:
      - name: Check PR for AI generation signals
        uses: actions/github-script@v7
        with:
          script: |
            const pr = context.payload.pull_request;
            const body = (pr.body || '').toLowerCase();
            const title = (pr.title || '').toLowerCase();
            
            const aiSignal1 = /this pr (introduces|adds|implements|fixes)/i;
            const aiSignal2 = /i have implemented/i;
            const aiSignal3 = /as an ai/i;
            const aiSignal4 = /ai assistant/i;
            const aiSignal5 = /language model/i;
            const aiSignal6 = /chatgpt/i;
            const aiSignal7 = /claude/i;
            const aiSignal8 = /gemini/i;
            const aiSignal9 = /generated by/i;
            const aiSignal10 = /ai-generated/i;
            const aiSignal11 = /i am an ai/i;
            const aiSignal12 = /i'm an ai/i;
            const aiSignal13 = /artificial intelligence/i;
            
            const combinedText = body + ' ' + title;
            let aiScore = 0;
            
            const signals = [aiSignal1, aiSignal2, aiSignal3, aiSignal4, aiSignal5, 
                           aiSignal6, aiSignal7, aiSignal8, aiSignal9, aiSignal10,
                           aiSignal11, aiSignal12, aiSignal13];
            
            signals.forEach(signal => {
              if (signal.test(combinedText)) {
                aiScore++;
              }
            });
            
            if (body.length < 50) {
              aiScore += 1;
            }
            
            if (aiScore >= 1) {
              const repo = context.repo;
              
              await github.rest.issues.addLabels({
                owner: repo.owner,
                repo: repo.repo,
                issue_number: pr.number,
                labels: ['needs-human-review', 'ai-detected']
              });
            }
